import { Card, CardGrid } from "@astrojs/starlight/components";

## Features

<CardGrid>
  <Card title="Automatic Resize Handling" icon="external">
    Automatically resizes Renderer, Camera, EffectComposer. Using the
    viewportResize event, you can easily set the resolution for custom shaders.
  </Card>
  <Card title="Smart Rendering" icon="rocket">
    Optimize performance by rendering frames only when necessary, reducing
    computational load.
  </Card>
  <Card title="Simplified Multiple Rendering" icon="laptop">
    Easily handle rendering for multiple scenes or viewports on a single canvas.
  </Card>
  <Card title="Object3D Property Binding" icon="puzzle">
    Simplify Object3D property management.
  </Card>
  <Card title="Event Programming" icon="comment">
    Add interactions to Object3D through programmable events, similar to DOM
    events. It is possible to bind events on changing position, scale, rotation,
    visibility and enabled state.
  </Card>
  <Card title="Focus and Blur" icon="magnifier">
    Enhance interactivity with focus and blur events.
  </Card>
  <Card title="Drag and Drop" icon="bars">
    Seamlessly integrate drag and drop functionality.
  </Card>
  <Card title="Hitbox Functionality" icon="close">
    Utilize hitboxes for customized intersections or simplified calculations.
  </Card>
  <Card title="Raycasting Options" icon="random">
    Choose between continuous or mouse movement-based raycasting, optimizing
    intersection operations.
  </Card>
  <Card title="Tweening" icon="right-arrow">
    Create smooth animations effortlessly with built-in tweening.
  </Card>
  <Card title="Simplified InstancedMesh" icon="approve-check">
    Manage InstancedMesh instances as if they were Object3D, simplifying
    creation and manipulation.
  </Card>
  <Card title="Asset Management" icon="add-document">
    Efficiently load and manage external assets and resources for your 3D
    projects.
  </Card>
</CardGrid>
